 }).addTo(map);
// BaseLayer
  const mapBaseLayer = {
       m_mono: "MIERUNE Mono",
      m_color: "MIERUNE Color"
 };
  // OverLayer
  const mapOverLayer = {
       o_std: "OpenStreetMap",
       t_pale: "GSI Pale",
       t_ort: "GSI Ort"
   };
 var BaseLayer = new L.ekmap.control.BaseLayer({
       baseLayers: mapBaseLayer,
       overLayers: mapOverLayer,
       opacityControl: true
  });
 map.addControl(BaseLayer,"top-left");
